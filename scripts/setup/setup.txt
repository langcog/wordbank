	echo -e '\n[clientdev]\nuser=wordbank\npassword = wordbank\ndatabase = wordbank\nhost = server.wordbank.stanford.edu\n\n[mysqld]\nskip-grant-tables' | sudo tee -a /etc/mysql/my.cnf
	sudo service mysql restart
	mysql -u root
	flush privileges;
	use mysql;
	update user set authentication_string=PASSWORD("") where User='root';
	update user set plugin="mysql_native_password"; # THIS LINE
	flush privileges;
	ALTER USER 'root'@'localhost' IDENTIFIED BY '';
	\q
	head -n -2 /etc/mysql/my.cnf  | sudo tee /etc/mysql/my.cnf
	sudo service mysql restart
	mysql -u root
	CREATE USER 'wordbank'@'localhost';
	CREATE USER 'ubuntu'@'localhost';
	CREATE USER ''@'localhost';
	GRANT ALL PRIVILEGES ON wordbank.* TO 'wordbank'@'localhost' WITH GRANT OPTION;
	GRANT ALL PRIVILEGES ON wordbank.* TO ''@'localhost' WITH GRANT OPTION;
	GRANT ALL PRIVILEGES ON *.* TO 'ubuntu'@'localhost' WITH GRANT OPTION;
	CREATE DATABASE wordbank;
	use wordbank;
	source wordbank.sql 
	\q
